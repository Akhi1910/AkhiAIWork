np.random.seed(45)

bandit_probabilities = [0.10, 0.40, 0.10, 0.10]

number_of_bandits = len(bandit_probabilities) # = n_actions
    
action_space = np.arange(number_of_bandits) # =[0,1,2,3]

number_of_trials = 20 
timesteps = 1000

arms = np.zeros(timesteps, dtype=int)

def step(action):
    rand = np.random.random()  # [0.0,1.0)
    reward = 1.0 if (rand < bandit_probabilities[action]) else 0.0
    return reward

def compute_regret(bandit_probabilities, arms, time_steps):
    probs = [bandit_probabilities[arm] for arm in arms]
    return np.cumsum(np.ones(time_steps)) * np.amax(bandit_probabilities) - np.cumsum(probs)


def cumulative_average_mean(r, n):
    return np.cumsum(r) / np.cumsum(np.ones(n))


def plot_rewards(reward_list, n):
    for r in reward_list:
        plt.plot(cumulative_average_mean(r[0], n), linewidth=3)
    plt.xlabel('Steps')
    plt.ylabel('Average reward')
    legend_str = [r[1] for r in reward_list]
    plt.legend(legend_str)
    plt.show()

def plot_regrets(regrets):

# Defining the parameters
number_of_bandits = len(bandit_probabilities) # as defined in the above code
action_space = np.arange(number_of_bandits)
number_of_trials = 20
timesteps = 1000                              
UCB_constants = [0.01, 1, 100]                # as requested in the question = c

# We can first define the UCB Policy function (that we take as UCB1)
def UCBpolicy(c, actions, q_values, iteration, number_of_invocations):
    # computing using the UCB1 formula - the reason I have chosen the UCB1 algorithm is because as it is described on Microsoft's website, it is specifically designed to tackle the computation of the bernoulli bandit problem. 
    # additionally, it is given to us in the question as 'UCB Action Selection'
    ucbpolicy = q_values + (c*np.sqrt(np.log(iteration)/number_of_invocations))
    # we then choose the action that gives us the highest UCB
    action = np.argmax(ucbpolicy)

# The following was for iterative purposes
    number_of_invocations[action] += 1 # As defined for 'Update N(a*)
    reward = step(actions[action])
    q_values[action] += ((((number_of_invocations[action]-1)/number_of_invocations[action]))*q_values[action]) + (reward/number_of_invocations[action])  # As defined for 'Update Qhat(a)'
    return q_values, number_of_invocations, action, reward

# create empty lists to be filled with each iteration
avg_rewards = []
avg_regrets = []

# iterate for the different values of c that we have defined : (0.1, 1, 100)
for c in UCB_constants:
    rewards = np.zeros((number_of_trials, timesteps))
    regrets = np.zeros((number_of_trials, timesteps))

    for i in range(number_of_trials):
        q_values = np.zeros(number_of_bandits)
        number_of_invocations = np.zeros(number_of_bandits)

        for t in range(timesteps):
            q_values, number_of_invocations, action, reward = UCBpolicy(c, action_space, q_values, t+1, number_of_invocations)
            arms[t] = action
            rewards[i, t] = reward
            regrets[i, t] = compute_regret(bandit_probabilities, arms[:t+1], t+1)[-1] 
            # using the above commands, we store the reward and regret

    # The avg_reward for this iteration (of c) is computed and then added to the list defined above using the append function
    avg_rewards.append((np.mean(rewards, axis=0), 'c='+str(c)))
    avg_regrets.append((np.mean(regrets, axis=0), 'c='+str(c)))

# Now the lists have been filled for average regrets and rewards, and it can be plotted using the commands defined in the given code.
# plot results
plot_rewards(avg_rewards, timesteps)
plot_regrets(avg_regrets)
